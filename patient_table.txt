                        <th>Phone</th>
                        <th>Blood Type</th>
                      </tr>
                    </thead>
                    <tbody>
                      {patients.map((p, i) => (
                        <tr key={p.uid || i}>
                          <td>{i + 1}</td>
                          <td>{p.name}</td>
                          <td>{p.email}</td>
                          <td>{p.gender || "—"}</td>
                          <td>{p.phone || "—"}</td>
                          <td>{p.bloodType || "—"}</td>
                        </tr>
                      ))}
                    </tbody>
                  </table>
                </div>
              </div>
            </div>
          )}
        </main>
      </div>
    </div>
  );
}
